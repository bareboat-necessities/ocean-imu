\newcommand{\safeplotinput}[1]{%
\IfFileExists{#1}{\resizebox{\linewidth}{!}{\input{#1}}}{%
\fbox{\begin{minipage}[c][0.55\linewidth][c]{0.92\linewidth}\centering
\textbf{Missing plot file}\\\texttt{#1}
\end{minipage}}%
}%
}

\newcommand{\specplots}[3]{%
    \begin{figure}[H]\centering
    \safeplotinput{spectrum_estimates_#1_#2_spectrum.pgf}
    \caption{#3 spectrum estimates at #2 sea state, overlaying classic and wavelets estimators against simulation reference.}
    \end{figure}
}

\newcommand{\HTplots}[3]{%
    \begin{figure}[H]\centering
    \safeplotinput{spectrum_estimates_#1_#2_hs.pgf}
    \caption{#3 $H_s$ estimates at #2 sea state for classic and wavelets estimators.}
    \end{figure}
    \begin{figure}[H]\centering
    \safeplotinput{spectrum_estimates_#1_#2_tp.pgf}
    \caption{#3 $T_p$ estimates at #2 sea state for classic and wavelets estimators.}
    \end{figure}
}
